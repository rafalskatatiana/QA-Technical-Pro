// Перший варіант з for
function drawTriangle(height, symbol) {
 for (let a = 1; a <= height; a++) {
  let row = "";
  for (let b = 1; b <= a; b++) {
   row += symbol;
  }
  console.log(row);
 }
}

drawTriangle(5, '*');
